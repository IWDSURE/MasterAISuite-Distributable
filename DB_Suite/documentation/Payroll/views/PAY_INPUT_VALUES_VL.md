## Query

```sql
SELECT pivf.INPUT_VALUE_ID INPUT_VALUE_ID, pivf.USER_ENTERABLE_FLAG USER_ENTERABLE_FLAG, pivf.EFFECTIVE_START_DATE EFFECTIVE_START_DATE, pivf.EFFECTIVE_END_DATE EFFECTIVE_END_DATE, pivf.ELEMENT_TYPE_ID ELEMENT_TYPE_ID, pivf.LOOKUP_TYPE LOOKUP_TYPE, pivf.FORMULA_ID FORMULA_ID, pivf.DISPLAY_SEQUENCE DISPLAY_SEQUENCE, pivf.GENERATE_DB_ITEMS_FLAG GENERATE_DB_ITEMS_FLAG, pivf.HOT_DEFAULT_FLAG HOT_DEFAULT_FLAG, pivf.MANDATORY_FLAG MANDATORY_FLAG, pivf.BASE_NAME BASE_NAME, pivf.UOM UOM, pivf.DEFAULT_VALUE DEFAULT_VALUE, pivf.MAX_VALUE MAX_VALUE, pivf.MIN_VALUE MIN_VALUE, pivf.VO_NAME VO_NAME, pivf.WARNING_OR_ERROR WARNING_OR_ERROR, pivf.USER_DISPLAY_FLAG USER_DISPLAY_FLAG, pivf.RESERVED_INPUT_VALUE RESERVED_INPUT_VALUE, pivf.RETRO_STATIC_FLAG RETRO_STATIC_FLAG, pivf.FORCE_RRV_MODE FORCE_RRV_MODE, pivf.VALIDATION_OVERRIDE_MESSAGE VALIDATION_OVERRIDE_MESSAGE, pivf.CONTEXT_ID CONTEXT_ID, pivftl.NAME NAME, pivf.LAST_UPDATE_DATE LAST_UPDATE_DATE, pivf.LAST_UPDATED_BY LAST_UPDATED_BY, pivf.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, pivf.CREATED_BY CREATED_BY, pivf.CREATION_DATE CREATION_DATE, pivf.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, pivf.ENTERPRISE_ID ENTERPRISE_ID, pivf.VALUE_SET_CODE VALUE_SET_CODE, pivf.SGUID SGUID FROM PAY_INPUT_VALUES_F pivf, PAY_INPUT_VALUES_TL pivftl WHERE pivftl.INPUT_VALUE_ID = pivf.INPUT_VALUE_ID AND pivftl.LANGUAGE = USERENV('LANG')
```
