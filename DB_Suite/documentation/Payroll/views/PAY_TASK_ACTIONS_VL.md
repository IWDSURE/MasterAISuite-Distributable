SELECT PAY_TASK_ACTIONS.TASK_ACTION_ID TASK_ACTION_ID, PAY_TASK_ACTIONS.BASE_TASK_ACTION_ID BASE_TASK_ACTION_ID, PAY_TASK_ACTIONS.BASE_TASK_ACTION_NAME BASE_TASK_ACTION_NAME, PAY_TASK_ACTIONS.LEGISLATIVE_DATA_GROUP_ID LEGISLATIVE_DATA_GROUP_ID, PAY_TASK_ACTIONS.LEGISLATION_CODE LEGISLATION_CODE, PAY_TASK_ACTIONS.ENTERPRISE_ID ENTERPRISE_ID, PAY_TASK_ACTIONS.EXECUTION_MODE EXECUTION_MODE, PAY_TASK_ACTIONS.ACTION_TYPE ACTION_TYPE, PAY_TASK_ACTIONS.ACTION_METHOD ACTION_METHOD, PAY_TASK_ACTIONS.ACTION_SEQUENCE ACTION_SEQUENCE, PAY_TASK_ACTIONS.BASE_TASK_ID BASE_TASK_ID, PAY_TASK_ACTIONS.DEFAULT_ACTION_FLAG, PAY_TASK_ACTIONS.PARENT_ACTION_FLAG, PAY_TASK_ACTIONS.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, PAY_TASK_ACTIONS.MODULE_ID MODULE_ID, PAY_TASK_ACTIONS.CREATED_BY CREATED_BY, PAY_TASK_ACTIONS.CREATION_DATE CREATION_DATE, PAY_TASK_ACTIONS.LAST_UPDATED_BY LAST_UPDATED_BY, PAY_TASK_ACTIONS.LAST_UPDATE_DATE LAST_UPDATE_DATE, PAY_TASK_ACTIONS.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, tl.ACTION_NAME ACTION_NAME, tl.DESCRIPTION DESCRIPTION, PAY_TASK_ACTIONS.SGUID SGUID, PAY_TASK_ACTIONS.ACTIVE_FLAG FROM PAY_TASK_ACTIONS, PAY_TASK_ACTIONS_TL tl WHERE PAY_TASK_ACTIONS.TASK_ACTION_ID = tl.TASK_ACTION_ID AND USERENV('LANG') = tl.LANGUAGE
