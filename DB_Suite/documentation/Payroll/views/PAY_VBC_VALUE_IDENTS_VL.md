## Query

```sql
SELECT pvi.VBC_VALUE_IDENT_ID, pvi.VBC_DEFINITION_ID, pvi.BASE_NAME, pvitl.NAME, pvitl.DESCRIPTION, pvitl.LANGUAGE, pvi.ALLOW_RANGE, pvi.CALC_TYPE_ID, pvi.REQUIRED_FLAG, pvi.ENTERABLE_FLAG, pvi.FLEXIBLE_FLAG, pvi.VBC_CRITERIA_ID, pvi.DATE_MODE, pvi.CURRENCY_CODE, pvi.UOM, pvi.PERIODICITY, pvi.PERIODICITY_TYPE, pvi.VALUE_SET_CODE, pvi.VALUE_SET_CODE2, pvi.VALUE_SET_CODE3, pvi.VO_NAME, pvi.VO_NAME2, pvi.VO_NAME3, pvi.ENTERPRISE_ID, pvi.LEGISLATIVE_DATA_GROUP_ID, pvi.LEGISLATION_CODE, pvi.MODULE_ID, pvi.OBJECT_VERSION_NUMBER, pvi.CREATION_DATE, pvi.CREATED_BY, pvi.LAST_UPDATED_BY, pvi.LAST_UPDATE_DATE, pvi.LAST_UPDATE_LOGIN, pvi.SEED_STATUS, pvi.SGUID, pvi.SEED_DATA_SOURCE FROM PAY_VBC_VALUE_IDENTS pvi, PAY_VBC_VALUE_IDENTS_TL pvitl WHERE pvi.VBC_VALUE_IDENT_ID = pvitl.VBC_VALUE_IDENT_ID AND pvitl.LANGUAGE = USERENV('LANG')
```
