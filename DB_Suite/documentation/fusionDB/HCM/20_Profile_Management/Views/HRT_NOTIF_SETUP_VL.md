## Query

```sql
SELECT NS.NOTIF_ID NOTIF_ID, NS.BUSINESS_GROUP_ID BUSINESS_GROUP_ID, NS.NOTIF_NAME NOTIF_NAME, NS.NOTIF_ENABLED NOTIF_ENABLED, NS.PRODUCT_CODE PRODUCT_CODE, NSTL.NOTIF_LABEL NOTIF_LABEL, NSTL.DESCRIPTION DESCRIPTION, NS.MODULE_ID MODULE_ID, NS.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, NS.CREATED_BY CREATED_BY, NS.CREATION_DATE CREATION_DATE, NS.LAST_UPDATED_BY LAST_UPDATED_BY, NS.LAST_UPDATE_DATE LAST_UPDATE_DATE, NS.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN FROM HRT_NOTIF_SETUP_B NS, HRT_NOTIF_SETUP_TL NSTL WHERE NS.NOTIF_ID = NSTL.NOTIF_ID AND NS.BUSINESS_GROUP_ID = NSTL.BUSINESS_GROUP_ID AND NSTL.LANGUAGE = USERENV('LANG')
```
