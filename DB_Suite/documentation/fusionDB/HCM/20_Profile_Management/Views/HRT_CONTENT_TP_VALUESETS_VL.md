## Query

```sql
SELECT CTV.CONTENT_VALUE_SET_ID CONTENT_VALUE_SET_ID, CTV.CONTENT_TYPE_ID CONTENT_TYPE_ID, CTV.CURATION_ENABLED_FLAG CURATION_ENABLED_FLAG, CTV.STATUS STATUS, CTV.BUSINESS_GROUP_ID BUSINESS_GROUP_ID, CTVTL.VALUE_SET_NAME VALUE_SET_NAME, CTVTL.VALUE_SET_DESCRIPTION VALUE_SET_DESCRIPTION, CTV.PARENT_CONTENT_VALUE_SET_ID PARENT_CONTENT_VALUE_SET_ID, CTV.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, CTV.CREATED_BY CREATED_BY, CTV.CREATION_DATE CREATION_DATE, CTV.LAST_UPDATE_DATE LAST_UPDATE_DATE, CTV.LAST_UPDATED_BY LAST_UPDATED_BY, CTV.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, CTV.MODULE_ID MODULE_ID, CTV.SEED_DATA_SOURCE SEED_DATA_SOURCE, CTV.CONTENT_VALUE_SET_CODE FROM HRT_CONTENT_TP_VALUESETS_B CTV, HRT_CONTENT_TP_VALUESETS_TL CTVTL WHERE CTV.CONTENT_VALUE_SET_ID = CTVTL.CONTENT_VALUE_SET_ID AND CTV.BUSINESS_GROUP_ID = CTVTL.BUSINESS_GROUP_ID AND CTVTL.LANGUAGE = USERENV('LANG')
```
