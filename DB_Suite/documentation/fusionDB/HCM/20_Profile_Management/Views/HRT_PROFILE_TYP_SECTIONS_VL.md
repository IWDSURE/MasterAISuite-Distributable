## Query

```sql
SELECT PTS.SECTION_ID, PTS.BUSINESS_GROUP_ID, PTSTL.NAME, PTSTL.DESCRIPTION, PTS.PROFILE_TYPE_ID, PTS.CONTENT_TYPE_ID, PTS.PARENT_SECTION_ID, PTS.CONTENT_TYPE_RELATION_ID, PTS.SECTION_ITEM_REQD_FLAG, PTS.APPROVAL_REQUIRED_FLAG, PTS.QUALIFIER_SET_ID1, PTS.QUALIFIER_SET_ID2, PTS.SECTION_CONTEXT, PTS.SECURITY_CONTEXT, PTS.DISPLAY_ORDER, PTS.SECTION_LAYOUT, PTS.SEED_DATA_SOURCE, PTS.TEMPLATE_FLAG, PTS.DFF_CONTEXT_NAME, PTS.DEFAULT_FLAG, PTS.ACTIVE_FLAG, PTS.ANALYTIC_PRIORITY, PTS.ANALYTIC_PRIORITY_SOURCE_CODE, PTS.OBJECT_VERSION_NUMBER, PTS.CREATED_BY, PTS.CREATION_DATE, PTS.LAST_UPDATED_BY, PTS.LAST_UPDATE_DATE, PTS.LAST_UPDATE_LOGIN, PTS.MODULE_ID, PTS.TEMPLATE_BASED_SECTION_FLAG, PTS.LINKEDIN_ENABLED_FLAG, PTS.SKILL_RECOMMEND_ENABLED_FLAG FROM HRT_PROFILE_TYP_SECTIONS PTS, HRT_PROFILE_TYP_SECTIONS_TL PTSTL WHERE PTS.SECTION_ID = PTSTL.SECTION_ID(+) AND PTS.BUSINESS_GROUP_ID = PTSTL.BUSINESS_GROUP_ID(+) AND PTSTL.LANGUAGE (+) = USERENV('LANG')
```
