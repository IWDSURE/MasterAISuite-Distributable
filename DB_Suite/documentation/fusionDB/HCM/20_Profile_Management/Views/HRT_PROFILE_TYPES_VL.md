## Query

```sql
SELECT PT.PROFILE_TYPE_ID, PT.BUSINESS_GROUP_ID, PTL.DESCRIPTION, PTL.SUMMARY, PT.PROFILE_TYPE_CODE, PT.PROFILE_STATUS_CODE, PT.PROFILE_USAGE_CODE, PT.DATE_FROM, PT.DATE_TO, PT.SYNC_FLEXFIELD_FLAG, PT.PROFILE_SYNC_FLAG, PT.SYNC_CRITICALITY_FLAG, PT.SYNC_DESCRIPTION_FLAG, PT.SYNC_QUALIFICATION_FLAG, PT.SYNC_RESPONSIBILITY_FLAG, PT.SYNC_RISK_FLAG, PT.OVERWRITE_SYNC_ITEM_FLAG, PT.SYNC_ATTACHMENT_FLAG, PT.PID_APPROVAL_FLAG, PT.MODULE_ID, PT.OBJECT_VERSION_NUMBER, PT.CREATED_BY, PT.CREATION_DATE, PT.LAST_UPDATED_BY, PT.LAST_UPDATE_DATE, PT.LAST_UPDATE_LOGIN, PT.AUTO_CREATE_FLAG, PT.SHOW_DESCRIPTION_FLAG, PT.SHOW_RESPONSIBILITY_FLAG, PT.SHOW_QUALIFICATION_FLAG, PT.SHOW_CRITICALITY_FLAG FROM HRT_PROFILE_TYPES_B PT, HRT_PROFILE_TYPES_TL PTL WHERE PT.PROFILE_TYPE_ID = PTL.PROFILE_TYPE_ID AND PT.BUSINESS_GROUP_ID = PTL.BUSINESS_GROUP_ID AND PTL.LANGUAGE = USERENV('LANG')
```
