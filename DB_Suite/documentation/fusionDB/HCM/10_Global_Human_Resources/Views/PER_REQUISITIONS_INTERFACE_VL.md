```sql
SELECT B.REQUISITION_INTERFACE_ID REQUISITION_INTERFACE_ID, B.REQUISITION_INTERFACE_CODE REQUISITION_INTERFACE_CODE, B.ENTERPRISE_ID ENTERPRISE_ID, TL.REQUISITION_TITLE REQUISITION_TITLE, B.INTERFACE_TYPE INTERFACE_TYPE, B.INTERFACE_SOURCE INTERFACE_SOURCE, B.INTERFACE_SOURCE_ID INTERFACE_SOURCE_ID, B.VACANCY_STATUS VACANCY_STATUS, B.REQUISITION_NUMBER REQUISITION_NUMBER, B.REQUISITION_STATUS REQUISITION_STATUS, B.REQUISITION_JUSTIFICATION REQUISITION_JUSTIFICATION, B.REQUISITION_TEMPLATE_CODE REQUISITION_TEMPLATE_CODE, B.TARGET_START_DATE TARGET_START_DATE, B.JOB_SCHEDULE JOB_SCHEDULE, B.TOTAL_OPEN_HEADCOUNT TOTAL_OPEN_HEADCOUNT, B.TOTAL_HIRED_HEADCOUNT TOTAL_HIRED_HEADCOUNT, B.TOTAL_AVAILABLE_HEADCOUNT TOTAL_AVAILABLE_HEADCOUNT, B.MANAGER_USER_NAME MANAGER_USER_NAME, B.MANAGER_ID MANAGER_ID, B.MANAGER_ASSIGNMENT_ID MANAGER_ASSIGNMENT_ID, B.MANAGER_TYPE MANAGER_TYPE, B.JOB_ID JOB_ID, B.ORGANIZATION_ID ORGANIZATION_ID, B.PRIMARY_LOCATION_ID PRIMARY_LOCATION_ID, B.POSITION_ID POSITION_ID, B.EMPLOYEE_STATUS EMPLOYEE_STATUS, B.UNLIMITED_HIRE UNLIMITED_HIRE, B.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, B.CREATED_BY CREATED_BY, B.CREATION_DATE CREATION_DATE, B.LAST_UPDATE_DATE LAST_UPDATE_DATE, B.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, B.LAST_UPDATED_BY LAST_UPDATED_BY FROM PER_REQUISITIONS_INTERFACE_B B, PER_REQUISITIONS_INTERFACE_TL TL WHERE B.REQUISITION_INTERFACE_ID = TL.REQUISITION_INTERFACE_ID AND TL.LANGUAGE = USERENV('LANG')
```
