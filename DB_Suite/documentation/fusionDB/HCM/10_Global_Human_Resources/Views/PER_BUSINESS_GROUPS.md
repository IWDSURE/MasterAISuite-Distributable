## Query

```sql
SELECT o.ORGANIZATION_ID BUSINESS_GROUP_ID, o.ORGANIZATION_ID, otl.NAME, o.EFFECTIVE_START_DATE, o.EFFECTIVE_END_DATE, o.INTERNAL_ADDRESS_LINE, o.LOCATION_ID, o2.ORG_INFORMATION1 DEFAULT_START_TIME, o2.ORG_INFORMATION2 DEFAULT_END_TIME, o2.ORG_INFORMATION3 WORKING_HOURS, o2.ORG_INFORMATION4 FREQUENCY, o3.ORG_INFORMATION1 SHORT_NAME, o3.ORG_INFORMATION2 METHOD_OF_GENERATION_EMP_NUM, o3.ORG_INFORMATION3 METHOD_OF_GENERATION_APL_NUM, o3.ORG_INFORMATION4 GRADE_STRUCTURE, o3.ORG_INFORMATION5 PEOPLE_GROUP_STRUCTURE, o3.ORG_INFORMATION6 JOB_STRUCTURE, o3.ORG_INFORMATION7 COST_ALLOCATION_STRUCTURE, o3.ORG_INFORMATION8 POSITION_STRUCTURE, o3.ORG_INFORMATION9 LEGISLATION_CODE, o3.ORG_INFORMATION10 CURRENCY_CODE, o3.ORG_INFORMATION14 SECURITY_GROUP_ID, o4.STATUS ENABLED_FLAG, o3.ORG_INFORMATION15 COMPETENCE_STRUCTURE, o3.ORG_INFORMATION16 METHOD_OF_GENERATION_CWK_NUM FROM HR_ALL_ORGANIZATION_UNITS_F o, HR_ORGANIZATION_UNITS_F_TL otl, HR_ORGANIZATION_INFORMATION_F o2, HR_ORGANIZATION_INFORMATION_F o3, HR_ORG_UNIT_CLASSIFICATIONS_F o4 WHERE o.ORGANIZATION_ID = otl.ORGANIZATION_ID AND o.ORGANIZATION_ID = o2.ORGANIZATION_ID(+) AND o.ORGANIZATION_ID = o3.ORGANIZATION_ID AND o.ORGANIZATION_ID = o4.ORGANIZATION_ID AND o3.ORG_INFORMATION_CONTEXT = 'PER_ENTERPRISE_INFO' AND o2.ORG_INFORMATION_CONTEXT(+) = 'PER_WORK_DAY_INFO' AND o4.CLASSIFICATION_CODE = 'ENTERPRISE' AND o4.STATUS = 'A' AND o.EFFECTIVE_START_DATE BETWEEN o4.EFFECTIVE_START_DATE AND o4.EFFECTIVE_END_DATE AND otl.LANGUAGE = USERENV('lang')
```
