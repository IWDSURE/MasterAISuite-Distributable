## Query

```sql
SELECT GRD.GRADE_ID, GRD.EFFECTIVE_START_DATE, GRD.EFFECTIVE_END_DATE, ASM.DETERMINANT_VALUE BUSINESS_UNIT, GRD.SET_ID, GRD.BUSINESS_GROUP_ID+0 BUSINESS_GROUP_ID, GRD.ACTION_OCCURRENCE_ID, GRD.PAY_SCALE_ID, GRD.STARTING_STEP, GRD.CEILING_STEP_ID, GRD.GRADE_CODE, GRD.GRADE_TYPE, GDT.NAME, GRD.ACTIVE_STATUS, GRD.ATTRIBUTE_CATEGORY, GRD.ATTRIBUTE1, GRD.ATTRIBUTE2, GRD.ATTRIBUTE3, GRD.ATTRIBUTE4, GRD.ATTRIBUTE5, GRD.ATTRIBUTE6, GRD.ATTRIBUTE7, GRD.ATTRIBUTE8, GRD.ATTRIBUTE9, GRD.ATTRIBUTE10, GRD.ATTRIBUTE11, GRD.ATTRIBUTE12, GRD.ATTRIBUTE13, GRD.ATTRIBUTE14, GRD.ATTRIBUTE15, GRD.ATTRIBUTE16, GRD.ATTRIBUTE17, GRD.ATTRIBUTE18, GRD.ATTRIBUTE19, GRD.ATTRIBUTE20, GRD.ATTRIBUTE21, GRD.ATTRIBUTE22, GRD.ATTRIBUTE23, GRD.ATTRIBUTE24, GRD.ATTRIBUTE25, GRD.ATTRIBUTE26, GRD.ATTRIBUTE27, GRD.ATTRIBUTE28, GRD.ATTRIBUTE29, GRD.ATTRIBUTE30, GRD.OBJECT_VERSION_NUMBER, GRD.CREATED_BY, GRD.CREATION_DATE, GRD.LAST_UPDATED_BY, GRD.LAST_UPDATE_DATE, GRD.LAST_UPDATE_LOGIN FROM PER_GRADES_F GRD, PER_GRADES_F_TL GDT, FND_SETID_ASSIGNMENTS ASM WHERE ASM.DETERMINANT_TYPE = 'BU' AND ASM.REFERENCE_GROUP_NAME = 'PER_GRADES' AND GRD.SET_ID = ASM.SET_ID AND GRD.GRADE_ID = GDT.GRADE_ID AND GRD.EFFECTIVE_START_DATE = GDT.EFFECTIVE_START_DATE AND GRD.EFFECTIVE_END_DATE = GDT.EFFECTIVE_END_DATE AND GDT.LANGUAGE = SYS_CONTEXT ('USERENV','LANG') AND TRUNC(SYSDATE) BETWEEN GRD.EFFECTIVE_START_DATE AND GRD.EFFECTIVE_END_DATE
```
