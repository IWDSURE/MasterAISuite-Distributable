## Query

```sql
SELECT HXT_TCLAYST_B.TCLAYST_ID TCLAYST_ID, HXT_TCLAYST_B.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, HXT_TCLAYST_B.ANS_SET_ID ANS_SET_ID, HXT_TCLAYST_B.TCSMR_SET_ID TCSMR_SET_ID, HXT_TCLAYST_B.LAYOUTSET_TYPE LAYOUTSET_TYPE, HXT_TCLAYST_TL.NAME NAME, HXT_TCLAYST_TL.DESCRIPTION DESCRIPTION, HXT_TCLAYST_B.SHORT_CODE SHORT_CODE, HXT_TCLAYST_B.OWNER OWNER, HXT_TCLAYST_B.COMPLETION_STATUS COMPLETION_STATUS, HXT_TCLAYST_B.AUDIT_USAGE_FLAG AUDIT_USAGE_FLAG, HXT_TCLAYST_B.DEFAULT_AUDIT_CONFIG_FLAG DEFAULT_AUDIT_CONFIG_FLAG, HXT_TCLAYST_B.MODULE_ID MODULE_ID, HXT_TCLAYST_B.CREATED_BY CREATED_BY, HXT_TCLAYST_B.CREATION_DATE CREATION_DATE, HXT_TCLAYST_B.LAST_UPDATED_BY LAST_UPDATED_BY, HXT_TCLAYST_B.LAST_UPDATE_DATE LAST_UPDATE_DATE, HXT_TCLAYST_B.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, HXT_TCLAYST_B.ENTERPRISE_ID ENTERPRISE_ID, HXT_TCLAYST_B.SGUID SGUID FROM HXT_TCLAYST_B HXT_TCLAYST_B, HXT_TCLAYST_TL HXT_TCLAYST_TL WHERE HXT_TCLAYST_B.TCLAYST_ID = HXT_TCLAYST_TL.TCLAYST_ID AND HXT_TCLAYST_TL.LANGUAGE = USERENV('LANG')
```
