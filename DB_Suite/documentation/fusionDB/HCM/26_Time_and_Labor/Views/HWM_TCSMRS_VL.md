```sql
SELECT HWM_TCSMRS_B.TCSMRS_ID TCSMRS_ID, HWM_TCSMRS_B.MODULE_ID MODULE_ID, HWM_TCSMRS_B.ENTERPRISE_ID ENTERPRISE_ID, HWM_TCSMRS_B.TCSMRS_CODE TCSMRS_CODE, HWM_TCSMRS_B.GRP_TYPE_ID GRP_TYPE_ID, HWM_TCSMRS_B.TM_ATRB_STRUCTURE_ID TM_ATRB_STRUCTURE_ID, HWM_TCSMRS_B.FUSION_APPLICATION FUSION_APPLICATION, HWM_TCSMRS_B.UPDATE_RAW_TM_SVC UPDATE_RAW_TM_SVC, HWM_TCSMRS_B.VALIDATE_RAW_TM_SVC VALIDATE_RAW_TM_SVC, HWM_TCSMRS_B.UPDATE_CALC_TM_SVC UPDATE_CALC_TM_SVC, HWM_TCSMRS_B.VALIDATE_CALC_TM_SVC VALIDATE_CALC_TM_SVC, HWM_TCSMRS_B.CREATED_BY CREATED_BY, HWM_TCSMRS_B.CREATION_DATE CREATION_DATE, HWM_TCSMRS_B.LAST_UPDATED_BY LAST_UPDATED_BY, HWM_TCSMRS_B.LAST_UPDATE_DATE LAST_UPDATE_DATE, HWM_TCSMRS_B.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, HWM_TCSMRS_TL.NAME NAME, HWM_TCSMRS_B.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, HWM_TCSMRS_B.SGUID SGUID FROM HWM_TCSMRS_B, HWM_TCSMRS_TL WHERE HWM_TCSMRS_B.TCSMRS_ID = HWM_TCSMRS_TL.TCSMRS_ID AND HWM_TCSMRS_TL.LANGUAGE = USERENV('LANG')
```
