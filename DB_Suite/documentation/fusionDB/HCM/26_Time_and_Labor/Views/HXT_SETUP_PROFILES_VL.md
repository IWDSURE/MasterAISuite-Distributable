## Query

```sql
SELECT HXT_SETUP_PROFILES_B.SETUP_PROFILE_ID SETUP_PROFILE_ID, HXT_SETUP_PROFILES_B.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, HXT_SETUP_PROFILES_B.ENTERPRISE_ID ENTERPRISE_ID, HXT_SETUP_PROFILES_B.MODULE_ID MODULE_ID, HXT_SETUP_PROFILES_TL.NAME NAME, HXT_SETUP_PROFILES_TL.DESCRIPTION DESCRIPTION, HXT_SETUP_PROFILES_B.PRECEDENCE PRECEDENCE, HXT_SETUP_PROFILES_B.PRODUCT_AREA PRODUCT_AREA, HXT_SETUP_PROFILES_B.CREATED_BY CREATED_BY, HXT_SETUP_PROFILES_B.CREATION_DATE CREATION_DATE, HXT_SETUP_PROFILES_B.LAST_UPDATED_BY LAST_UPDATED_BY, HXT_SETUP_PROFILES_B.LAST_UPDATE_DATE LAST_UPDATE_DATE, HXT_SETUP_PROFILES_B.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, HXT_SETUP_PROFILES_TL.ENTERPRISE_ID ENTERPRISE_ID1, HXT_SETUP_PROFILES_B.SETUP_PROFILE_CD SETUP_PROFILE_CD, HXT_SETUP_PROFILES_B.SGUID SGUID, HXT_SETUP_PROFILES_B.STATUS STATUS FROM HXT_SETUP_PROFILES_B, HXT_SETUP_PROFILES_TL WHERE HXT_SETUP_PROFILES_B.SETUP_PROFILE_ID = HXT_SETUP_PROFILES_TL.SETUP_PROFILE_ID AND HXT_SETUP_PROFILES_TL.LANGUAGE = USERENV('LANG')
```
