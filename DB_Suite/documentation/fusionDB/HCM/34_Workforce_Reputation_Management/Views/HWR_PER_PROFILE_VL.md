## Query

```sql
SELECT B.PRFL_ID, B.SOURCE_ID, B.FIRST_NAME, B.MIDDLE_NAME, B.LAST_NAME, B.ALIAS, B.GENDER_CODE, B.EMAIL, B.IS_ACTIVE_FLAG, B.RESUME_URL, B.IMAGE_URL, B.ABOUT_ME, B.ASSOCIATIONS, B.CURRENT_STATUS, B.DNA_CHECKSUM, B.GEEKCODE, B.HEADLINE, B.HONORS, B.INDUSTRY, B.INTERESTS, B.LANGUAGES, B.MYERS_BRIGG, B.NICKNAME, B.PERSONAL_URL, B.PLAN, B.SKILLS, B.SPECIALTIES, B.TITLE, B.PROFILE_REPO, B.PROFILE_TYPE, B.URL, B.USER_ID, B.USERNAME, B.USER_GUID, B.USER_DISTINGUISHED_NAME, B.MULTITENANCY_USERNAME, B.CAREER_STATEMENT, B.JOB_NAME, B.COMPANY_NAME, T.STREET_ADDRESS, T.PHONE, B.CREATED_BY, B.CREATION_DATE, B.LAST_UPDATE_LOGIN, B.LAST_UPDATED_BY, B.LAST_UPDATE_DATE FROM HWR_PER_PROFILE_B B, HWR_PER_PROFILE_TL T WHERE B.PRFL_ID = T.PRFL_ID AND B.SOURCE_ID = T.SOURCE_ID AND T.LANGUAGE = USERENV('LANG')
```
