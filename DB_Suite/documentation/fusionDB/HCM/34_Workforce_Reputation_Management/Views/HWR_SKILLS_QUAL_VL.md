## Query

```sql
SELECT HRT_PROFILE_ITEMS.PROFILE_ITEM_ID , HRT_PROFILE_ITEMS.BUSINESS_GROUP_ID , HRT_PROFILE_ITEMS.PROFILE_ID , HRT_PROFILE_ITEMS.CONTENT_TYPE_ID, HRT_PROFILE_ITEMS.CONTENT_ITEM_ID, HRT_PROFILE_ITEMS.ITEM_TEXT240_2, HRT_PROFILE_ITEMS.ITEM_TEXT240_3, HRT_PROFILE_ITEMS.ITEM_TEXT240_4, HRT_PROFILE_ITEMS.ITEM_TEXT2000_1, HRT_PROFILE_ITEMS.ITEM_TEXT30_1, HRT_PROFILE_ITEMS.ITEM_TEXT30_2, HRT_PROFILE_ITEMS.ITEM_TEXT30_3, HRT_PROFILE_ITEMS.ITEM_TEXT30_7, HRT_PROFILE_ITEMS.ITEM_TEXT30_8, HRT_CONTENT_TYPES_B.CONTEXT_NAME, HRT_PROFILE_ITEMS.CREATED_BY, HRT_PROFILE_ITEMS.CREATION_DATE, HRT_PROFILE_ITEMS.LAST_UPDATED_BY, HRT_PROFILE_ITEMS.LAST_UPDATE_DATE, HRT_PROFILES_B.PERSON_ID, HRT_PROFILES_B.PARTY_ID, HRT_CONTENT_ITEMS_TL.NAME FROM HRT_PROFILE_ITEMS, HRT_PROFILES_B, HRT_CONTENT_TYPES_B,HRT_CONTENT_ITEMS_TL WHERE HRT_PROFILES_B.PROFILE_ID = HRT_PROFILE_ITEMS.PROFILE_ID AND HRT_PROFILE_ITEMS.CONTENT_TYPE_ID = HRT_CONTENT_TYPES_B.CONTENT_TYPE_ID AND HRT_PROFILE_ITEMS.BUSINESS_GROUP_ID = HRT_CONTENT_TYPES_B.BUSINESS_GROUP_ID AND HRT_PROFILE_ITEMS.BUSINESS_GROUP_ID=HRT_CONTENT_ITEMS_TL.BUSINESS_GROUP_ID AND HRT_PROFILE_ITEMS.CONTENT_ITEM_ID=HRT_CONTENT_ITEMS_TL.CONTENT_ITEM_ID AND HRT_PROFILES_B.PROFILE_USAGE_CODE = 'P' AND HRT_CONTENT_TYPES_B.CONTEXT_NAME IN ('CERTIFICATION','COMPETENCY','DEGREE','EDUCATION_LEVEL','HONOR','LANGUAGE','MEMBERSHIP') AND HRT_CONTENT_ITEMS_TL.LANGUAGE = SYS_CONTEXT('USERENV', 'LANG') UNION SELECT HRT_PROFILE_ITEMS.PROFILE_ITEM_ID , HRT_PROFILE_ITEMS.BUSINESS_GROUP_ID , HRT_PROFILE_ITEMS.PROFILE_ID , HRT_PROFILE_ITEMS.CONTENT_TYPE_ID, HRT_PROFILE_ITEMS.CONTENT_ITEM_ID, HRT_PROFILE_ITEMS.ITEM_TEXT240_2, HRT_PROFILE_ITEMS.ITEM_TEXT240_3, HRT_PROFILE_ITEMS.ITEM_TEXT240_4, HRT_PROFILE_ITEMS.ITEM_TEXT2000_1, HRT_PROFILE_ITEMS.ITEM_TEXT30_1, HRT_PROFILE_ITEMS.ITEM_TEXT30_2, HRT_PROFILE_ITEMS.ITEM_TEXT30_3, HRT_PROFILE_ITEMS.ITEM_TEXT30_7, HRT_PROFILE_ITEMS.ITEM_TEXT30_8, HRT_CONTENT_TYPES_B.CONTEXT_NAME, HRT_PROFILE_ITEMS.CREATED_BY, HRT_PROFILE_ITEMS.CREATION_DATE, HRT_PROFILE_ITEMS.LAST_UPDATED_BY, HRT_PROFILE_ITEMS.LAST_UPDATE_DATE, HRT_PROFILES_B.PERSON_ID, HRT_PROFILES_B.PARTY_ID, HRT_PROFILE_ITEMS.ITEM_TEXT240_1 AS NAME FROM HRT_PROFILE_ITEMS, HRT_PROFILES_B,HRT_CONTENT_TYPES_B WHERE HRT_PROFILES_B.PROFILE_ID = HRT_PROFILE_ITEMS.PROFILE_ID AND HRT_PROFILE_ITEMS.CONTENT_TYPE_ID = HRT_CONTENT_TYPES_B.CONTENT_TYPE_ID AND HRT_PROFILE_ITEMS.BUSINESS_GROUP_ID = HRT_CONTENT_TYPES_B.BUSINESS_GROUP_ID AND HRT_PROFILES_B.PROFILE_USAGE_CODE = 'P' AND HRT_PROFILE_ITEMS.ITEM_TEXT240_1 IS NOT NULL AND HRT_CONTENT_TYPES_B.CONTEXT_NAME IN ('AREA_OF_STUDY','SPECIAL_PROJECT','PREV_EMPLOY','HIGHEST_EDUCATION_LEVEL')
```
