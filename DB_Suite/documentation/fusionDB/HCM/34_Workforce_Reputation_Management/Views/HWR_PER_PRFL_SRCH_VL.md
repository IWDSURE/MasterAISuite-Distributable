## Query

```sql
SELECT PRFL.PRFL_ID, PRFL.FIRST_NAME, PRFL.MIDDLE_NAME, PRFL.LAST_NAME, PRFL.ALIAS, PRFL.EMAIL, PRFL.GENDER_CODE, PRFL.SOURCE_ID, PRFL.IS_ACTIVE_FLAG, PRFL.RESUME_URL, PRFL.IMAGE_URL, PRFL.ABOUT_ME, PRFL.ASSOCIATIONS, PRFL.CURRENT_STATUS, PRFL.DNA_CHECKSUM, PRFL.GEEKCODE, PRFL.HEADLINE, PRFL.HONORS, PRFL.INDUSTRY, PRFL.INTERESTS, PRFL.LANGUAGES, PRFL.MYERS_BRIGG, PRFL.NICKNAME, PRFL.PERSONAL_URL, PRFL.PLAN, PRFL.SKILLS, PRFL.SPECIALTIES, PRFL.TITLE, PRFL.PROFILE_REPO, PRFL.PROFILE_TYPE, PRFL.URL, PRFL.USER_ID, PRFL.USERNAME, PRFL.USER_GUID, PRFL.USER_DISTINGUISHED_NAME, PRFL.MULTITENANCY_USERNAME, PRFL.CAREER_STATEMENT, PRFL.JOB_NAME, PRFL.COMPANY_NAME, PRFL.STREET_ADDRESS, PRFL.PHONE, PRFL.CREATED_BY, PRFL.CREATION_DATE, PRFL.LAST_UPDATE_LOGIN, PRFL.LAST_UPDATED_BY, PRFL.LAST_UPDATE_DATE, GBL_PRFL.GBL_PRFL_ID GBL_PRFL_ID, GBL_PRFL.NAME GBL_PRFL_NAME, SRC.NAME SRC_NAME FROM HWR_PER_PROFILE_VL PRFL JOIN HWR_SRC_SOURCE_VL SRC ON (PRFL.SOURCE_ID = SRC.SOURCE_ID) LEFT JOIN HWR_PER_GBL_PRFL_PRFL_X XREF ON (PRFL.PRFL_ID = XREF.PRFL_ID) LEFT JOIN HWR_PER_GBL_PRFL_VL GBL_PRFL ON (XREF.GBL_PRFL_ID = GBL_PRFL.GBL_PRFL_ID)
```
