## Query

```sql
select ACTIVITY_ID, ACTIVITY_MEASURE_TYPE, VALUE, UNITS, CREATED_BY, LAST_UPDATE_DATE, LAST_UPDATED_BY, LAST_UPDATE_LOGIN from HWR_ACT_MSR UNION select ACTIVITY_ID, 'BMR_RATE' ACTIVITY_MEASURE_TYPE, VALUE, UNITS, CREATED_BY, LAST_UPDATE_DATE, LAST_UPDATED_BY, LAST_UPDATE_LOGIN from HWR_ACT_MSR_BMR UNION select ACTIVITY_ID, 'DISTANCE' ACTIVITY_MEASURE_TYPE, VALUE, UNITS, CREATED_BY, LAST_UPDATE_DATE, LAST_UPDATED_BY, LAST_UPDATE_LOGIN from HWR_ACT_MSR_DISTANCE UNION select ACTIVITY_ID, 'CALORIES' ACTIVITY_MEASURE_TYPE, VALUE, UNITS, CREATED_BY, LAST_UPDATE_DATE, LAST_UPDATED_BY, LAST_UPDATE_LOGIN from HWR_ACT_MSR_CALORIE UNION select ACTIVITY_ID, 'ACTIVITY_TIME_HIGH' ACTIVITY_MEASURE_TYPE, VALUE, UNITS, CREATED_BY, LAST_UPDATE_DATE, LAST_UPDATED_BY, LAST_UPDATE_LOGIN from HWR_ACT_MSR_ACTIVE UNION select ACTIVITY_ID, 'ACTIVITY_TIME_MODERATE' ACTIVITY_MEASURE_TYPE, VALUE, UNITS, CREATED_BY, LAST_UPDATE_DATE, LAST_UPDATED_BY, LAST_UPDATE_LOGIN from HWR_ACT_MSR_MD UNION select ACTIVITY_ID, 'ACTIVITY_TIME_LOW' ACTIVITY_MEASURE_TYPE, VALUE, UNITS, CREATED_BY, LAST_UPDATE_DATE, LAST_UPDATED_BY, LAST_UPDATE_LOGIN from HWR_ACT_MSR_LW UNION select ACTIVITY_ID, 'STEPS' ACTIVITY_MEASURE_TYPE, VALUE, UNITS, CREATED_BY, LAST_UPDATE_DATE, LAST_UPDATED_BY, LAST_UPDATE_LOGIN from HWR_ACT_MSR_STEPS
```
