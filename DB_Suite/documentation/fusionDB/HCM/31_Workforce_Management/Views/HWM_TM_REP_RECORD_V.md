## Query

```sql
SELECT TR.TM_REC_ID, TR.TM_REC_VERSION, TR.TM_REC_TYPE, TR.MEASURE, TR.UNIT_OF_MEASURE, TR.START_TIME, TR.STOP_TIME, TR.COMMENT_TEXT, TR.RESOURCE_ID, TR.SUBRESOURCE_ID, TR.ACTIVITY_IN_ID, TR.ACTIVITY_OUT_ID, TR.GRP_TYPE_ID, TR.CREATION_DATE, TR.ORDER_ENTERED, TR.ACTUAL_DATE, TC.TM_REC_GRP_ID, TCD.START_TIME DAY_START_TIME FROM HWM_TM_REC_GRP_USAGES TRU ,HWM_TM_REC TR ,HWM_TM_REC_GRP TCD ,HWM_TM_REC_GRP TC WHERE TC.LATEST_VERSION = 'Y' AND NVL(TC.DELETE_FLAG,'N') = 'N' and TC.GRP_TYPE_ID = 100 AND TCD.GRP_TYPE_ID = 101 AND TR.GRP_TYPE_ID = 102 AND TC.TM_REC_GRP_ID = TCD.PARENT_TM_REC_GRP_ID AND TC.TM_REC_GRP_VERSION = TCD.PARENT_TM_REC_GRP_VERSION AND TCD.TM_REC_GRP_ID = TRU.TM_REC_GRP_ID AND TCD.TM_REC_GRP_VERSION = TRU.TM_REC_GRP_VERSION AND TRU.TM_REC_ID = TR.TM_REC_ID AND TRU.TM_REC_VERSION = TR.TM_REC_VERSION AND TCD.LATEST_VERSION = 'Y' AND TR.LATEST_VERSION = 'Y' AND TRU.LATEST_VERSION = 'Y' AND NVL(TCD.DELETE_FLAG,'N') = 'N' AND NVL(TR.DELETE_FLAG,'N') = 'N' UNION SELECT TR.TM_REC_ID, TR.TM_REC_VERSION, TR.TM_REC_TYPE, TR.MEASURE, TR.UNIT_OF_MEASURE, TR.START_TIME, TR.STOP_TIME, TR.COMMENT_TEXT, TR.RESOURCE_ID, TR.SUBRESOURCE_ID, TR.ACTIVITY_IN_ID, TR.ACTIVITY_OUT_ID, TR.GRP_TYPE_ID, TR.CREATION_DATE, TR.ORDER_ENTERED, TR.ACTUAL_DATE, TC.TM_REC_GRP_ID, TRUNC(TR.START_TIME) DAY_START_TIME FROM HWM_TM_REC TR ,HWM_TM_REC_GRP TC WHERE TC.LATEST_VERSION = 'Y' AND NVL(TC.DELETE_FLAG,'N') = 'N' and TC.GRP_TYPE_ID = 100 AND TR.START_TIME >= TC.START_TIME - 1 AND TR.START_TIME <= TC.STOP_TIME + 1 AND NVL( TR.actual_date,TR.START_TIME) >= TC.START_TIME AND NVL( TR.actual_date,TR.START_TIME) <= TC.STOP_TIME AND TR.GRP_TYPE_ID = 106 AND TC.RESOURCE_ID = TR.RESOURCE_ID AND TR.LATEST_VERSION = 'Y' AND NVL(TR.DELETE_FLAG,'N') = 'N'
```
