## Query

```sql
SELECT HWM_GRPS_B.GRP_ID, HWM_GRPS_B.LIST_ID, HWM_GRPS_B.EVAL_SOURCE_CD, HWM_GRPS_B.APPL_USG_CD, HWM_GRPS_B.APPLIES_TO_CD, HWM_GRPS_B.FREEZE_FLAG, HWM_GRPS_B.RUN_TYPE, HWM_GRPS_B.EVAL_STATUS_CD, HWM_GRPS_B.NUM_DAYS_PREV, HWM_GRPS_B.NUM_DAYS_NEXT, HWM_GRPS_B.LAST_REFRESH_DT, HWM_GRPS_B.NEXT_REFRESH_DT, HWM_GRPS_B.OBJECT_VERSION_NUMBER, HWM_GRPS_B.CREATED_BY, HWM_GRPS_B.CREATION_DATE, HWM_GRPS_B.LAST_UPDATE_DATE, HWM_GRPS_B.LAST_UPDATED_BY, HWM_GRPS_B.LAST_UPDATE_LOGIN, HWM_GRPS_B.ENTERPRISE_ID ENTERPRISE_ID, HWM_GRPS_B.MODULE_ID, HWM_GRPS_TL.GROUP_NAME, HWM_GRPS_TL.DESCRIPTION, HWM_GRPS_B.GRP_CODE, HWM_GRPS_B.ONLINE_EVAL_FLAG, HWM_GRPS_B.ASSIGNMENT_TO_USE_CD ASSIGNMENT_TO_USE_CD, HWM_GRPS_B.SGUID SGUID, HWM_GRPS_B.MASS_EVAL_FLAG, HWM_GRPS_B.MEM_TYPE_CD , HWM_GRPS_B.INCLUDE_SUSPENDED FROM HWM_GRPS_B, HWM_GRPS_TL WHERE HWM_GRPS_B.GRP_ID = HWM_GRPS_TL.GRP_ID AND HWM_GRPS_TL.LANGUAGE = USERENV('LANG')
```
