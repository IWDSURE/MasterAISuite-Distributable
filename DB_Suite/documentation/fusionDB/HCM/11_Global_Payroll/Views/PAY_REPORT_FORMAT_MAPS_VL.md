```sql
SELECT prfmf.REPORT_TYPE REPORT_TYPE, prfmf.REPORT_QUALIFIER REPORT_QUALIFIER, prfmf.RANGE_CODE RANGE_CODE, prfmf.ASSIGNMENT_ACTION_CODE ASSIGNMENT_ACTION_CODE, prfmf.REPORT_CATEGORY REPORT_CATEGORY, prfmf.UPDATABLE_FLAG UPDATABLE_FLAG, prfmf.REPORT_FORMAT_MAPPING_ID REPORT_FORMAT_MAPPING_ID, prfmf.LEGISLATIVE_DATA_GROUP_ID LEGISLATIVE_DATA_GROUP_ID, prfmf.LEGISLATION_CODE LEGISLATION_CODE, prfmf.ENTERPRISE_ID ENTERPRISE_ID, prfmf.TEMPORARY_ACTION_FLAG TEMPORARY_ACTION_FLAG, prfmtl.DISPLAY_NAME DISPLAY_NAME, prfmf.EFFECTIVE_START_DATE EFFECTIVE_START_DATE, prfmf.EFFECTIVE_END_DATE EFFECTIVE_END_DATE, prfmf.LAST_UPDATED_BY LAST_UPDATED_BY, prfmf.LAST_UPDATE_DATE LAST_UPDATE_DATE, prfmf.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, prfmf.CREATED_BY CREATED_BY, prfmf.CREATION_DATE CREATION_DATE, prfmf.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, prfmf.MODULE_ID module_id, prfmf.SGUID SGUID FROM PAY_REPORT_FORMAT_MAPS_F prfmf, PAY_REPORT_FORMAT_MAPS_TL prfmtl WHERE prfmtl.REPORT_FORMAT_MAPPING_ID = prfmf.REPORT_FORMAT_MAPPING_ID AND prfmtl.LANGUAGE = USERENV('LANG')
```
