## Query

```sql
SELECT pbgu.BAL_GRP_USAGE_ID BAL_GRP_USAGE_ID, pbgu.BASE_GROUP_USAGE_NAME BASE_GROUP_USAGE_NAME, pbgutl.GROUP_USAGE_NAME GROUP_USAGE_NAME, pbgutl.DESCRIPTION DESCRIPTION, pbgu.ENTERPRISE_ID ENTERPRISE_ID, pbgu.LEGISLATION_CODE LEGISLATION_CODE, pbgu.LEGISLATIVE_DATA_GROUP_ID LEGISLATIVE_DATA_GROUP_ID, pbgu.MODULE_ID MODULE_ID, pbgu.BALANCE_GROUP_ID BALANCE_GROUP_ID, pbgu.BALANCE_REPORT_TYPE BALANCE_REPORT_TYPE, pbgu.FORMAT_TYPE FORMAT_TYPE, pbgu.REPORT_SORT_TYPE_ID REPORT_SORT_TYPE_ID, pbgu.LAST_UPDATE_DATE LAST_UPDATE_DATE, pbgu.LAST_UPDATED_BY LAST_UPDATED_BY, pbgu.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, pbgu.CREATED_BY CREATED_BY, pbgu.CREATION_DATE CREATION_DATE, pbgu.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER FROM PAY_BAL_GRP_USAGES pbgu, PAY_BAL_GRP_USAGES_TL pbgutl WHERE pbgutl.BAL_GRP_USAGE_ID = pbgu.BAL_GRP_USAGE_ID AND pbgutl.LANGUAGE = USERENV('LANG')
```
