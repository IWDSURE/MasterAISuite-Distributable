## Query

```sql
SELECT pao.ALLOW_OVERRIDES_ID ALLOW_OVERRIDES_ID, pao.EFFECTIVE_START_DATE EFFECTIVE_START_DATE, pao.EFFECTIVE_END_DATE EFFECTIVE_END_DATE, paotl.PROMPT PROMPT, pao.VALUE_DEFN_ID VALUE_DEFN_ID, pao.TYPE TYPE, pao.LEGISLATIVE_DATA_GROUP_ID LEGISLATIVE_DATA_GROUP_ID, pao.LEGISLATION_CODE LEGISLATION_CODE, pao.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, pao.ENTERPRISE_ID ENTERPRISE_ID, pao.MODULE_ID MODULE_ID, pao.CREATED_BY CREATED_BY, pao.CREATION_DATE CREATION_DATE, pao.LAST_UPDATE_DATE LAST_UPDATE_DATE, pao.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, pao.LAST_UPDATED_BY LAST_UPDATED_BY, pao.SEED_STATUS SEED_STATUS, pao.SGUID SGUID FROM PAY_ALLOW_OVERRIDES_F pao, PAY_ALLOW_OVERRIDES_TL paotl WHERE pao.ALLOW_OVERRIDES_ID = paotl.ALLOW_OVERRIDES_ID AND paotl.LANGUAGE = USERENV('LANG')
```
