## Query

```sql
SELECT b.TASK_PARAMETER_ID, b.BASE_TASK_PARAMETER_ID, b.BASE_TASK_ACTION_ID, b.BASE_TASK_PARAMETER_NAME, b.MODULE_ID, b.OBJECT_VERSION_NUMBER, b.DISPLAY_FLAG, b.PARAM_DISP_TYPE, b.PARAM_USAGE_TYPE, b.PARAM_SEQUENCE, b.PARAM_LOOKUP, b.PARAM_VALUE_SET, b.DEFAULT_TYPE, b.DEFAULT_VAL, b.LEGISLATIVE_DATA_GROUP_ID, b.LEGISLATION_CODE, b.ENTERPRISE_ID, b.DATA_TYPE, b.DATA_DOMAIN, b.NAMESPACE, b.REPETITIVE_FLAG, b.PARAMETER_TYPE, b.ELEMENT_NAME, b.PARENT_TASK_PARAM_ID, b.DRIVING_LABEL_METHOD, b.LAST_UPDATE_DATE, b.LAST_UPDATED_BY, b.LAST_UPDATE_LOGIN, b.CREATED_BY, b.CREATION_DATE, b.PARAM_LEVEL, tl.PARAMETER_NAME, tl.DESCRIPTION, b.SGUID SGUID, b.HDL_ATTRIBUTE, b.PARAMETER_CODE, b.UI_EXPRESSION, b.BASE_CONTEXT_NAME, b.MULTI_VALUED_FLAG FROM PAY_TASK_PARAMETERS b, PAY_TASK_PARAMETERS_TL tl WHERE b.TASK_PARAMETER_ID = tl.TASK_PARAMETER_ID AND USERENV('LANG') = tl.LANGUAGE
```
