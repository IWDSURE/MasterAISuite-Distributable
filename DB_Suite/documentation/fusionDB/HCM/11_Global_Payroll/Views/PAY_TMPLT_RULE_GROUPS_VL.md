## Query

```sql
SELECT PTRG.RULE_GROUP_ID RULE_GROUP_ID, PTRG.BASE_RULE_GROUP_NAME BASE_RULE_GROUP_NAME, PTRGTL.GROUP_TEXT GROUP_TEXT, PTRG.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, PTRG.LAST_UPDATE_DATE LAST_UPDATE_DATE, PTRG.LAST_UPDATED_BY LAST_UPDATED_BY, PTRG.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, PTRG.CREATED_BY CREATED_BY, PTRG.CREATION_DATE CREATION_DATE, PTRG.GROUP_LOCATION, PTRG.TEMPLATE_ID, PTRG.LEGISLATIVE_DATA_GROUP_ID, PTRG.LEGISLATION_CODE, PTRG.ENTERPRISE_ID, PTRG.MODULE_ID, PTRG.SGUID SGUID FROM PAY_TMPLT_RULE_GROUPS PTRG, PAY_TMPLT_RULE_GROUPS_TL PTRGTL WHERE PTRG.RULE_GROUP_ID = PTRGTL.RULE_GROUP_ID AND PTRGTL.LANGUAGE = USERENV('LANG')
```
