## Query

```sql
SELECT B.CAMPAIGN_ID, B.CAMPAIGN_CODE, TL.CAMPAIGN_NAME, TL.CAMPAIGN_DESCRIPTION, B.CAMPAIGN_STATUS_CODE, B.PURPOSE_CODE, B.TARGET_GOAL_VALUE, B.CAMPAIGN_LANGUAGE_CODE, B.CAMPAIGN_TYPE_CODE, B.SEARCH_REQUISITIONS_FLAG, B.REQ_SAVED_SEARCH_ID, B.AUDIENCE_SAVED_SEARCH_ID, B.AUDIENCE_POPULATED_FLAG, B.AUD_REHIRE_ELIG_ENABLED_FLAG, B.AUD_EMAIL_LIMIT_ENABLED_FLAG, B.ACTIVATED_ON_DATE, B.CANCELLED_ON_DATE, B.CLOSED_ON_DATE, B.OBJECT_VERSION_NUMBER, B.CREATED_BY, B.CREATION_DATE, B.LAST_UPDATED_BY, B.LAST_UPDATE_DATE, B.LAST_UPDATE_LOGIN, B.VANITY_CONFIG_ID, B.JOURNEY_ID, B.ELIG_PRFL_ID, B.ELIG_OBJ_ID, B.TEAM_ID, B.JOURNEY_COMMUNICATION_GOAL, B.JOURNEY_ASSIGNED_DATE, B.EXCLUDE_JOURNEY_AUD_FLAG, B.CAMPAIGN_DURATION_CODE, B.RESPONSE_OPTION, B.AUDIENCE_COUNT, B.HCM_LIST_ID, B.HCM_LIST_EXTRACTION_DATE, B.CAMPAIGN_MEDIUM_CODE FROM IRC_CAMPAIGNS_B B, IRC_CAMPAIGNS_TL TL WHERE B.CAMPAIGN_ID = TL.CAMPAIGN_ID AND TL.LANGUAGE = USERENV('LANG')
```
