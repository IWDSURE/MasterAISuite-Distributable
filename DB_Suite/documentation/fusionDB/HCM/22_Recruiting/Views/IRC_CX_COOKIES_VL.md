```sql
SELECT IRC_CX_COOKIES_B.COOKIE_ID, IRC_CX_COOKIES_B.OBJECT_STATUS, IRC_CX_COOKIES_B.OBJECT_VERSION_NUMBER, IRC_CX_COOKIES_B.SITE_ID, IRC_CX_COOKIES_B.IS_ENABLED_COOKIE_MESSAGES, IRC_CX_COOKIES_B.IS_ENABLED_DECLINE_BUTTON, IRC_CX_COOKIES_B.CREATED_BY, IRC_CX_COOKIES_B.CREATION_DATE, IRC_CX_COOKIES_B.LAST_UPDATED_BY, IRC_CX_COOKIES_B.LAST_UPDATE_DATE, IRC_CX_COOKIES_B.LAST_UPDATE_LOGIN, IRC_CX_COOKIES_B.IS_ENABLED_COOKIE_PREFERENCES, IRC_CX_COOKIES_B.IS_ENABLED_CUSTOM_COOKIE, IRC_CX_COOKIES_TL.POP_UP_MESSAGE, IRC_CX_COOKIES_TL.POLICY_MESSAGE, IRC_CX_COOKIES_TL.ACCEPT_BUTTON_LABEL, IRC_CX_COOKIES_TL.COOKIE_LINK_LABEL, IRC_CX_COOKIES_TL.PREFERENCES_BUTTON_LABEL, IRC_CX_COOKIES_TL.PREFERENCES_SAVE_BUTTON_LABEL, IRC_CX_COOKIES_TL.STRICT_CATEGORY_MESSAGE, IRC_CX_COOKIES_TL.FUNCTIONAL_CATEGORY_MESSAGE, IRC_CX_COOKIES_TL.ANALYTICAL_CATEGORY_MESSAGE, IRC_CX_COOKIES_TL.CUSTOM_CATEGORY_BUTTON_LABEL, IRC_CX_COOKIES_TL.CUSTOM_CATEGORY_DESCRIPTION, IRC_CX_COOKIES_TL.DECLINE_BUTTON_LABEL FROM IRC_CX_COOKIES_B, IRC_CX_COOKIES_TL WHERE IRC_CX_COOKIES_B.COOKIE_ID = IRC_CX_COOKIES_TL.COOKIE_ID AND IRC_CX_COOKIES_TL.LANGUAGE = USERENV('LANG')
```
