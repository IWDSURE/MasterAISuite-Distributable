## Query

```sql
SELECT TNJOBCERTIFICATIONSB.TN_JOB_CERTIFICATION_ID, TNJOBCERTIFICATIONSB.TN_JOB_ID, TNJOBCERTIFICATIONSB.FA_CERTIFICATION_ID, TNJOBCERTIFICATIONSB.FA_SECTION_ID, TNJOBCERTIFICATIONSB.FA_CONTENT_ITEM_ID, TNJOBCERTIFICATIONSB.FA_CONTENT_ITEM_CODE, TNJOBCERTIFICATIONSB.FA_ESTABLISHMENT_ID, TNJOBCERTIFICATIONSB.FA_ESTABLISHMENT_CODE, TNJOBCERTIFICATIONSB.FA_COUNTRY_CODE, TNJOBCERTIFICATIONSB.FA_STATE_PROVINCE_CODE, TNJOBCERTIFICATIONSB.REQUIRED_FLAG, TNJOBCERTIFICATIONSTL.SECTION_NAME, TNJOBCERTIFICATIONSTL.CERTIFICATION_NAME, TNJOBCERTIFICATIONSTL.ESTABLISHMENT_NAME, TNJOBCERTIFICATIONSB.OBJECT_VERSION_NUMBER, TNJOBCERTIFICATIONSB.CREATED_BY, TNJOBCERTIFICATIONSB.CREATION_DATE, TNJOBCERTIFICATIONSB.LAST_UPDATED_BY, TNJOBCERTIFICATIONSB.LAST_UPDATE_DATE, TNJOBCERTIFICATIONSB.LAST_UPDATE_LOGIN FROM IRC_TN_JOB_CERTIFICATIONS_B TNJOBCERTIFICATIONSB, IRC_TN_JOB_CERTIFICATIONS_TL TNJOBCERTIFICATIONSTL WHERE TNJOBCERTIFICATIONSB.TN_JOB_CERTIFICATION_ID = TNJOBCERTIFICATIONSTL.TN_JOB_CERTIFICATION_ID AND TNJOBCERTIFICATIONSTL.LANGUAGE = USERENV('LANG')
```
