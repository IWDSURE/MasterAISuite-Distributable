## Query

```sql
SELECT PROFILE_ITEM_ID AS WORK_PREF_ID, BUSINESS_GROUP_ID, PROFILE_ID, PARENT_PROFILE_ITEM_ID, SECTION_ID, CONTENT_TYPE_ID, CONTENT_ITEM_ID , ITEM_TEXT30_1 AS NAT_TRAVEL_REQUIRED_FLAG, ITEM_TEXT30_3 AS INTL_TRAVEL_REQUIRED_FLAG, ITEM_TEXT30_2 AS NAT_TRAVEL_FREQUENCY, ITEM_TEXT30_4 AS INTL_TRAVEL_FREQUENCY, ITEM_TEXT30_14 AS WORK_DAYS, ITEM_TEXT30_15 AS WORK_HOURS, ITEM_TEXT30_5 AS RELOCATE_FLAG, ITEM_TEXT30_12 AS PAY_RANGE, ITEM_DECIMAL_1 AS FULL_TIME_EQUIVALENT, ITEM_NUMBER_1 AS WORK_MONTHS, ITEM_NUMBER_2 AS WORK_YEARS, IMPORTANCE AS IMPORTANCE, DATE_FROM, DATE_TO, OBJECT_VERSION_NUMBER , CREATED_BY , CREATION_DATE , LAST_UPDATED_BY , LAST_UPDATE_DATE , LAST_UPDATE_LOGIN FROM HRT_PROFILE_ITEMS PI WHERE CONTENT_TYPE_ID = 117 AND SECTION_ID IN (SELECT SECTION_ID FROM IRC_PROFILE_CONT_SECTIONS_V PTS WHERE PTS.PROFILE_TYPE_ID = 2 AND PTS.CONTENT_TYPE_ID = 117 AND PTS.BUSINESS_GROUP_ID = PI.BUSINESS_GROUP_ID)
```
