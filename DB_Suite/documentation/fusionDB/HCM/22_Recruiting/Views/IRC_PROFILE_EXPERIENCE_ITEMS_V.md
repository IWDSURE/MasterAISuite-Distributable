## Query

```sql
SELECT PROFILE_ITEM_ID AS PREV_EMPLOYMENT_ID , BUSINESS_GROUP_ID , PROFILE_ID , PARENT_PROFILE_ITEM_ID , SECTION_ID , CONTENT_TYPE_ID , CONTENT_ITEM_ID , ITEM_TEXT240_1 AS EMPLOYER_NAME , ITEM_TEXT240_9 AS JOB_TITLE , ITEM_DATE_1 AS START_DATE , ITEM_DATE_2 AS END_DATE , ITEM_CLOB_2 AS RESPONSIBILITIES , ITEM_TEXT240_12 AS EMPLOYER_CITY , STATE_PROVINCE_ID AS EMPLOYER_STATE_ID , COUNTRY_ID AS EMPLOYER_COUNTRY_ID , ITEM_TEXT30_1 AS CURRENT_JOB_FLAG , ITEM_TEXT30_9 AS FULL_TIME_CODE , ITEM_TEXT30_5 AS INTERNAL_FLAG , ITEM_TEXT240_13 AS BUSINESS_TYPE , ITEM_TEXT240_11 AS EMPLOYER_PHONE , ITEM_TEXT240_10 AS DEPARTMENT , ITEM_NUMBER_8 AS JOB_FAMILY_ID , ITEM_TEXT30_7 AS JOB_FUNCTION , ITEM_TEXT240_2 AS OTHER_JOB_FUNCTION , NVL(ITEM_CLOB_3,TO_CLOB(ITEM_TEXT2000_1)) AS ACHIEVEMENTS , ITEM_NUMBER_1 AS DIRECT_REPORTS , ITEM_TEXT240_8 AS STARTING_POSITION , ITEM_TEXT240_6 AS SUPERVISOR_NAME , ITEM_TEXT240_7 AS SUPERVISOR_TITLE , ITEM_TEXT240_4 AS SUPERVISOR_EMAIL , ITEM_TEXT240_3 AS SUPERVISOR_PHONE , ITEM_TEXT30_2 AS CONTACT_SUPERVISOR_FLAG , ITEM_CLOB_4 AS LEAVING_REASON , ITEM_CLOB_1 AS ADDITIONAL_INFO , ITEM_DECIMAL_1 AS START_COMPENSATION , ITEM_DECIMAL_2 AS END_COMPENSATION , ITEM_TEXT240_14 AS OTHER_COMPENSATION , ITEM_TEXT2000_2 AS COMPANY_URL , DATE_FROM, DATE_TO, OBJECT_VERSION_NUMBER , CREATED_BY , CREATION_DATE , LAST_UPDATED_BY , LAST_UPDATE_DATE , LAST_UPDATE_LOGIN, STATE_PROVINCE_CODE AS STATE_PROVINCE_CODE , COUNTRY_CODE AS COUNTRY_CODE FROM HRT_PROFILE_ITEMS PI WHERE CONTENT_TYPE_ID = 129 AND SECTION_ID IN (SELECT SECTION_ID FROM IRC_PROFILE_CONT_SECTIONS_V PTS WHERE PTS.PROFILE_TYPE_ID = 1 AND PTS.CONTENT_TYPE_ID = 129 AND PTS.BUSINESS_GROUP_ID = PI.BUSINESS_GROUP_ID)
```
