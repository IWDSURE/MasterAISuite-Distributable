## Query

```sql
SELECT PROFILE_ITEM_ID AS WORK_PREF_ID, BUSINESS_GROUP_ID, PROFILE_ID, PARENT_PROFILE_ITEM_ID, SECTION_ID, CONTENT_TYPE_ID, CONTENT_ITEM_ID , ITEM_TEXT30_1 AS NAT_TRAVEL_FLAG , ITEM_TEXT30_4 AS NAT_TRAVEL_FREQUENCY , ITEM_TEXT30_3 AS INTL_TRAVEL_FLAG , ITEM_TEXT30_2 AS INTL_TRAVEL_FREQUENCY , ITEM_TEXT30_5 AS RELOCATE_FLAG , ITEM_TEXT30_6 AS ALL_LOCATIONS_FLAG , ITEM_NUMBER_1 AS PREFERRED_LOCATION_ID1 , ITEM_NUMBER_2 AS PREFERRED_LOCATION_ID2 , ITEM_NUMBER_3 AS PREFERRED_LOCATION_ID3 , ITEM_NUMBER_4 AS PREFERRED_LOCATION_ID4 , ITEM_NUMBER_5 AS EXCLUDED_LOCATION_ID1 , ITEM_NUMBER_6 AS EXCLUDED_LOCATION_ID2 , ITEM_NUMBER_7 AS EXCLUDED_LOCATION_ID3 , ITEM_NUMBER_8 AS EXCLUDED_LOCATION_ID4 , ITEM_NUMBER_10 AS RELOCATE_DURATION , ITEM_TEXT240_1 AS RELOCATION_REASON , ITEM_CLOB_1 AS COMMENTS , ITEM_TEXT30_7 AS TEMP_ASSIGN_FLAG , ITEM_TEXT30_8 AS PART_ASSIGN_FLAG , ITEM_TEXT30_9 AS FLEXIBLE_WORK_FLAG , ITEM_NUMBER_9 AS MINIMUM_PAY , ITEM_TEXT30_10 AS CURRENCY_CODE , ITEM_TEXT30_11 AS PAY_FREQUENCY , ITEM_TEXT30_12 AS PAY_RANGE , ITEM_DECIMAL_1 AS FULL_TIME_EQUIVALENT , ITEM_TEXT30_14 AS WORK_DAYS , ITEM_TEXT30_15 AS WORK_HOURS , ITEM_TEXT30_13 AS INTL_TRAVEL_REQUIRED_FLAG , IMPORTANCE AS IMPORTANCE, DATE_FROM, DATE_TO, OBJECT_VERSION_NUMBER , CREATED_BY , CREATION_DATE , LAST_UPDATED_BY , LAST_UPDATE_DATE , LAST_UPDATE_LOGIN FROM HRT_PROFILE_ITEMS PI WHERE CONTENT_TYPE_ID = 117 AND SECTION_ID IN (SELECT SECTION_ID FROM IRC_PROFILE_CONT_SECTIONS_V PTS WHERE PTS.PROFILE_TYPE_ID = 1 AND PTS.CONTENT_TYPE_ID = 117 AND PTS.BUSINESS_GROUP_ID = PI.BUSINESS_GROUP_ID)
```
