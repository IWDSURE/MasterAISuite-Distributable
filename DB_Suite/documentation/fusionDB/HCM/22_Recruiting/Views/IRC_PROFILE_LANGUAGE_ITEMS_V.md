## Query

```sql
SELECT PROFILE_ITEM_ID AS LANGUAGE_ID, BUSINESS_GROUP_ID, PROFILE_ID, PARENT_PROFILE_ITEM_ID, SECTION_ID, CONTENT_TYPE_ID, CONTENT_ITEM_ID AS CONTENT_ITEM_ID , ITEM_TEXT240_1 AS LANGUAGE_NAME , RATING_LEVEL_ID2 AS WRITING_LEVEL_ID , RATING_LEVEL_ID3 AS SPEAKING_LEVEL_ID , RATING_LEVEL_ID1 AS READING_LEVEL_ID , ITEM_TEXT30_1 AS NATIVE_SPEAKER_FLAG , ITEM_TEXT30_2 AS ABLE_TO_TRANSLATE_FLAG , ITEM_TEXT30_3 AS ABLE_TO_TEACH_FLAG , ITEM_DATE_1 AS EVALUATED_ON , ITEM_TEXT30_4 AS EVALUATION_TYPE , ITEM_TEXT240_2 AS EVALUATION_LOCATION_ID , ITEM_CLOB_1 AS COMMENTS , RATING_MODEL_ID1 AS READING_MODEL_ID , RATING_MODEL_ID2 AS WRITING_MODEL_ID , RATING_MODEL_ID3 AS SPEAKING_MODEL_ID, IMPORTANCE AS IMPORTANCE, DATE_FROM, DATE_TO, OBJECT_VERSION_NUMBER , CREATED_BY , CREATION_DATE , LAST_UPDATED_BY , LAST_UPDATE_DATE , LAST_UPDATE_LOGIN FROM HRT_PROFILE_ITEMS PI WHERE CONTENT_TYPE_ID = 109 AND SECTION_ID IN (SELECT SECTION_ID FROM IRC_PROFILE_CONT_SECTIONS_V PTS WHERE PTS.PROFILE_TYPE_ID = 1 AND PTS.CONTENT_TYPE_ID = 109 AND PTS.BUSINESS_GROUP_ID = PI.BUSINESS_GROUP_ID)
```
