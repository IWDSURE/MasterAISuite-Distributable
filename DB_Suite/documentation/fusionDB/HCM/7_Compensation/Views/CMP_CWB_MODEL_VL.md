## Query

```sql
SELECT B.MODEL_ID, T.NAME, B.DATE_SAVED, B.PLAN_ID, B.PERIOD_ID, B.COMPONENT_ID, B.PERSON_ID, B.PERSON_EVENT_ID, B.CRIT_CD1, B.CRIT_CD2, B.CRIT_CD3, B.CRIT_CD4, B.ALCT_BY_CD, B.VAL_TYPE_CD, B.ACCESS_CD, B.REFERENCE_ONLY, B.SHOW_SUMMARY_FLAG, B.SHOW_PROPERTIES_FLAG, B.OBJECT_VERSION_NUMBER, B.LAST_UPDATE_DATE, B.LAST_UPDATED_BY, B.LAST_UPDATE_LOGIN, B.CREATED_BY, B.CREATION_DATE, B.ASSIGNMENT_ID, B.BUSINESS_GROUP_ID, B.SELECT_PURPOSE, B.BUDGET_FUND_LEVEL, B.OVERWRITE_BUDGETS_FLAG, B.PUBLISH_LEVEL_FLAG , B.IS_BATCH_MODE, B.LOGIN_PERSON_ID, B.ROLE FROM CMP_CWB_MODEL_TL T, CMP_CWB_MODEL_B B WHERE B.MODEL_ID = T.MODEL_ID AND T.LANGUAGE = USERENV('LANG') AND B.BUSINESS_GROUP_ID = T.BUSINESS_GROUP_ID
```
