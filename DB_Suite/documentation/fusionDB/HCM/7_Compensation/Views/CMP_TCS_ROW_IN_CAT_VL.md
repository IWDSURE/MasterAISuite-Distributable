## Query

```sql
Select CAT.ROW_IN_CAT_ID ,CAT.CAT_ID ,CAT.SUBCAT_ID ,CAT.ORDR_NUM ,CAT_TL.DISPLAY_NAME ,CAT.URL ,CAT_TL.DESCRIPTION ,CAT.ZERO_VALUE_OPTIONS ,CAT.HIDE_ZERO_ALERT_FLAG ,CAT_TL.ZERO_ALERT_MSG ,CAT.EXCLUDE_FROM_SMRY_FLAG ,CAT.STMT_ID ,CAT.BUSINESS_GROUP_ID ,CAT.CREATED_BY ,CAT.CREATION_DATE ,CAT.LAST_UPDATED_BY ,CAT.LAST_UPDATE_DATE ,CAT.LAST_UPDATE_LOGIN ,CAT.OBJECT_VERSION_NUMBER FRom CMP_TCS_ROW_IN_CAT CAT , CMP_TCS_ROW_IN_CAT_TL CAT_TL WHERE CAT.ROW_IN_CAT_ID = CAT_TL.ROW_IN_CAT_ID AND CAT_TL.LANGUAGE = USERENV('LANG') AND CAT.BUSINESS_GROUP_ID = CAT_TL.BUSINESS_GROUP_ID
```
