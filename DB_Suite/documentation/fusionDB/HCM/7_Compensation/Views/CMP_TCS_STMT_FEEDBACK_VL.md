## Query

```sql
SELECT FEEDBACK.STMT_FEEDBACK_ID , FEEDBACK.STMT_ID , FEEDBACK.STMT_PERD_ID , FEEDBACK.PERD_RUN_ID , FEEDBACK.QUESTION1_REQUIRED_FLAG , FEEDBACK.QUESTION2_REQUIRED_FLAG , FEEDBACK.QUESTION3_REQUIRED_FLAG , FEEDBACK.QUESTION4_REQUIRED_FLAG , FEEDBACK.QUESTION5_REQUIRED_FLAG , FEEDBACK_TL.LANGUAGE , FEEDBACK_TL.SOURCE_LANG , FEEDBACK_TL.RATING_INSTRUCTION , FEEDBACK_TL.QUESTION1 , FEEDBACK_TL.QUESTION2 , FEEDBACK_TL.QUESTION3 , FEEDBACK_TL.QUESTION4 , FEEDBACK_TL.QUESTION5 , FEEDBACK.REQUEST_ID , FEEDBACK.JOB_DEFINITION_NAME , FEEDBACK.JOB_DEFINITION_PACKAGE , FEEDBACK.CREATED_BY , FEEDBACK.CREATION_DATE , FEEDBACK.LAST_UPDATE_DATE , FEEDBACK.LAST_UPDATED_BY , FEEDBACK.LAST_UPDATE_LOGIN FROM CMP_TCS_STMT_FEEDBACK_B FEEDBACK, CMP_TCS_STMT_FEEDBACK_TL FEEDBACK_TL WHERE FEEDBACK.STMT_FEEDBACK_ID = FEEDBACK_TL.STMT_FEEDBACK_ID AND FEEDBACK_TL.LANGUAGE = USERENV('LANG')
```
