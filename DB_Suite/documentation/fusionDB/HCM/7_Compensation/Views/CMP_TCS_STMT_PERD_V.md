## Query

```sql
select PERD.STMT_PERD_ID ,PERD.STMT_ID ,PERD_TL.DISPLAY_PERIOD ,PERD.START_DATE ,PERD.END_DATE ,PERD.AVAILABILITY_DATE ,PERD.XCHG_RATE_DATE ,PERD.HIDE_WELCOME_FLAG ,PERD_TL.WELCOME_TITLE ,PERD_TL.WELCOME_TEXT ,PERD.STMT_GENERATED_FLAG ,PERD.FEEDBACK_ENABLED_FLAG ,PERD.PERD_FEEDBACK_FLAG ,PERD.STMT_PERSON_COUNT ,PERD.STMT_DATA_COUNT ,PERD.STMT_XML_COUNT ,PERD.BUSINESS_GROUP_ID ,PERD_TL.BUSINESS_GROUP_ID BUSINESS_GROUP_ID2 ,PERD.CREATED_BY ,PERD.CREATION_DATE ,PERD.LAST_UPDATED_BY ,PERD.LAST_UPDATE_DATE ,PERD.LAST_UPDATE_LOGIN ,PERD.OBJECT_VERSION_NUMBER ,PERD_TL.LANGUAGE FROM CMP_TCS_STMT_PERD PERD , CMP_TCS_STMT_PERD_TL PERD_TL WHERE PERD.STMT_PERD_ID = PERD_TL.STMT_PERD_ID AND PERD.BUSINESS_GROUP_ID = PERD_TL.BUSINESS_GROUP_ID
```
