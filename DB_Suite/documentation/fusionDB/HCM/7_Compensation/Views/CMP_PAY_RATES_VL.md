## Query

```sql
SELECT T.RATE_NAME, T.RATE_DISPLAY_NAME, T.DESCRIPTION, T.LANGUAGE, T.SOURCE_LANG, B.PAY_RATE_ID, B.BUSINESS_GROUP_ID, B.LEGISLATIVE_DATA_GROUP_ID, B.RATE_STATUS, B.AVAILABLE_FROM_DATE, B.PAY_RATE_CODE, B.RATE_GROUP_CODE, B.REPRESENTS_BASE, B.ADDS_TO_OVERALL_PAY, B.RATE_TYPE, B.AMOUNT_ROUNDING_CODE, B.AMOUNT_SCALE, B.PERCENT_SCALE, B.POINT_SCALE, B.DEFAULT_TYPE, B.DEFAULT_SOURCE, B.DEFAULT_FORMULA_ID, B.DEFAULT_GRADE_RATE_ID, B.DEFAULT_AMOUNT, B.DEFAULT_PERCENT, B.DEFAULT_POINTS, B.CURRENCY_SOURCE_CODE, B.CURRENCY_CODE, B.FREQUENCY_SOURCE, B.FREQUENCY_CODE, B.ANNUALIZATION_FACTOR, B.CHANGE_DEFAULT_AMOUNT, B.CHANGE_DEFAULT_PERCENT, B.CHANGE_DEFAULT_POINTS, B.VALUE_REQUIRED, B.ZERO_IF_VALUE_BLANK, B.LOWER_LIMIT, B.UPPER_LIMIT, B.VALIDATION_FORMULA_ID, B.IGNORE_LIMITS_CHECK, B.LIMIT_VIOLATION_BEHAVIOR, B.ELEMENT_TYPE_ID, B.INPUT_VALUE_ID, B.ENTRY_CREATION, B.OBJECT_VERSION_NUMBER, B.CREATED_BY, B.CREATION_DATE, B.LAST_UPDATED_BY, B.LAST_UPDATE_DATE, B.LAST_UPDATE_LOGIN FROM CMP_PAY_RATES_TL T, CMP_PAY_RATES_B B WHERE B.PAY_RATE_ID = T.PAY_RATE_ID AND T.LANGUAGE = USERENV('LANG') AND T.BUSINESS_GROUP_ID = B.BUSINESS_GROUP_ID
```
