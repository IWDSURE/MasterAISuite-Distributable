## Query

```sql
SELECT CAT.CAT_ID, CAT_TL.NAME, CAT.CAT_TYPE_CD, CAT_TL.ADDL_INFO, CAT_TL.ADDL_INFO_TITLE, CAT_TL.CHART1_TITLE, CAT.CHART1_TYPE_CD, CAT_TL.CHART2_TITLE, CAT.CHART2_TYPE_CD, CAT.CHART_RN_ORDR_NUM, CAT_TL.CHART_RN_TITLE, CAT.CONTRIBUTIONS_TYPE_CD, CAT_TL.DESC_TXT, CAT.DESC_TXT_ORDR_NUM, CAT.HIDE_ADDL_INFO_FLAG, CAT.HIDE_CHART_RN_FLAG, CAT.HIDE_DESC_TXT_FLAG, CAT.HISTORY_DATA_FLAG, CAT.LEVEL_OF_DETAIL, CAT.NNMNTRY_UOM, CAT.NUMBER_OF_COLUMNS, CAT.STAT_CD, CAT.STOCK_GRANT_TYPE_CD, CAT.STOCK_TRADING_SYMBOL, CAT.TABLE_ORDR_NUM, CAT.BUSINESS_GROUP_ID, CAT.CREATED_BY, CAT.CREATION_DATE, CAT.LAST_UPDATED_BY, CAT.LAST_UPDATE_DATE, CAT.LAST_UPDATE_LOGIN, CAT.OBJECT_VERSION_NUMBER, CAT.HIDE_TABLE_FLAG, CAT.VALID_CHART1_FLAG, CAT.VALID_CHART2_FLAG , cat.HIDE_OL_ITEM_DETAILS_CD, cat.HIDE_PDF_ITEM_DETAILS_CD, cat.HIDE_CALC_STOCK_BTN_FLAG FROM CMP_TCS_CAT CAT, CMP_TCS_CAT_TL CAT_TL WHERE CAT.CAT_ID = CAT_TL.CAT_ID AND CAT_TL.LANGUAGE = USERENV('LANG') AND CAT.BUSINESS_GROUP_ID = CAT_TL.BUSINESS_GROUP_ID
```
