```sql
SELECT B.LAYOUT_ATTRIBUTE_ID, B.LAYOUT_ID, B.SEQUENCE_NUMBER, B.BUSINESS_OBJECT_ID, B.ATTRIBUTE_NAME, TL.ATTRIBUTE_LABEL, TL.DESCRIPTION, B.ATTRIBUTE_TYPE, B.ATTRIBUTE_SUB_CATEGORY, B.ATTRIBUTE_DATATYPE, B.MANDATORY_FLAG, B.VISIBLE_FLAG, B.USER_KEY_FLAG, B.PARENT_USER_KEY_FLAG, B.SOURCE_ATTRIBUTE_ID, B.LIST_TYPE, B.DISPLAY_TYPE, B.LIST_VIEW_DEF_NAME, B.DISPLAY_ATTR_LIST, B.RETURN_ATTR_NAME, B.LIST_VIEW_CRIT_LIST, B.LIST_BIND_MAP, B.LIST_QUERY_CRITERIA, B.FLEXFIELD_CODE, B.FLEXFIELD_CONTEXT_CODE, B.OBJECT_VERSION_NUMBER, B.CREATED_BY, B.CREATION_DATE, B.LAST_UPDATED_BY, B.LAST_UPDATE_DATE, B.LAST_UPDATE_LOGIN, B.STAGE_COL_NAME, B.DEFAULT_VALUE, B.ENABLE_FLAG, B.READ_ONLY, B.DEFAULT_VAL_DATE, B.DEFAULT_VAL_TMSTMP, B.ENTERPRISE_ID, B.LAYOUT_BUS_OBJ_ID, B.GROUP_CODE, B.INTKEY_HINT_ALLOWED_FLAG, B.ATTRIBUTE_HINT, B.HINTS, B.ATTRIBUTE_SUBTYPE, B.PARENT_BUSINESS_OBJECT_ID, B.PARENT_ATTRIBUTE_NAME, B.USER_KEY_ATTR_NAMES, B.SECURE_FLAG, B.DATA_LENGTH, B.OBSOLETE_FLAG, B.MODULE_ID, B.DEPENDENT_ATTR_NAMES, B.ATTRIBUTE_KEY, B.CUSTOM_QUERY_CRITERIA, B.CUSTOM_VIEW_CRITERIA, B.CUSTOM_QUERY_HINTS, B.CUSTOM_VIEW_HINTS, B.DEFAULT_EXPRESSION, B.KEY_PREFERENCE, B.SEED_DATA_SOURCE, B.ATTRIB_KEY_REF, B.ATTRIB_INFO1, B.ATTRIB_INFO2, B.SGUID FROM HRC_DL_LAYOUT_ATTRIBUTES_B B, HRC_DL_LAYOUT_ATTRIBUTES_TL TL WHERE B.LAYOUT_ATTRIBUTE_ID = TL.LAYOUT_ATTRIBUTE_ID AND TL.LANGUAGE = USERENV('LANG')
```
