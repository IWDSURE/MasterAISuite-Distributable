```sql
SELECT A.ID ID, A.WELCOME_IMAGE_ID WELCOME_IMAGE_ID, A.EINREF_START_DATE EINREF_START_DATE, A.EINREF_START_TIME EINREF_START_TIME, A.EINREF_FREQUENCY EINREF_FREQUENCY, A.EINREF_NEXT_RUN_DATE EINREF_NEXT_RUN_DATE, A.EINREF_NEXT_RUN_TIME EINREF_NEXT_RUN_TIME, A.CREATED_BY CREATED_BY, A.CREATION_DATE CREATION_DATE, A.LAST_UPDATED_BY LAST_UPDATED_BY, A.LAST_UPDATE_DATE LAST_UPDATE_DATE, A.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, B.WELCOME_MESSAGE WELCOME_MESSAGE, B.WAIVER_AGREEMENT WAIVER_AGREEMENT, B.WAIVER_TITLE WAIVER_TITLE, A.EIN_VALIDATION_ON EIN_VALIDATION_ON, A.IRS_URL IRS_URL, B.USER_AGREEMENT USER_AGREEMENT, A.WELCOME_IMAGE_URL WELCOME_IMAGE_URL, A.ADMIN_ROLE ADMIN_ROLE, A.PHOTO_APPROVAL_SETTING PHOTO_APPROVAL_SETTING, A.PHOTO_AGREEMENT_SETTING PHOTO_AGREEMENT_SETTING FROM HHR_VLTR_SETTINGS_B A, HHR_VLTR_SETTINGS_TL B WHERE A.ID = B.ID AND B.LANGUAGE = USERENV('LANG')
```
