```sql
SELECT HRA_FEEDBACK_REQ_TMPLS_B.FEEDBACK_REQ_TMPL_ID FEEDBACK_REQ_TMPL_ID, HRA_FEEDBACK_REQ_TMPLS_B.BUSINESS_GROUP_ID BUSINESS_GROUP_ID, HRA_FEEDBACK_REQ_TMPLS_B.STATUS_CODE STATUS_CODE, HRA_FEEDBACK_REQ_TMPLS_B.QUESTIONNAIRE_ID QUESTIONNAIRE_ID, HRA_FEEDBACK_REQ_TMPLS_B.STANDARD_ALERT_DAYS STANDARD_ALERT_DAYS, HRA_FEEDBACK_REQ_TMPLS_B.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, HRA_FEEDBACK_REQ_TMPLS_B.CREATION_DATE CREATION_DATE, HRA_FEEDBACK_REQ_TMPLS_B.CREATED_BY CREATED_BY, HRA_FEEDBACK_REQ_TMPLS_B.LAST_UPDATED_BY LAST_UPDATED_BY, HRA_FEEDBACK_REQ_TMPLS_B.LAST_UPDATE_DATE LAST_UPDATE_DATE, HRA_FEEDBACK_REQ_TMPLS_B.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, HRA_FEEDBACK_REQ_TMPLS_TL.NAME NAME, HRA_FEEDBACK_REQ_TMPLS_TL.COMMENTS COMMENTS, HRA_FEEDBACK_REQ_TMPLS_B.INCLUDE_IN_PERF_DOC_FLAG INCLUDE_IN_PERF_DOC_FLAG, HRA_FEEDBACK_REQ_TMPLS_B.TEMPLATE_TYPE TEMPLATE_TYPE, HRA_FEEDBACK_REQ_TMPLS_B.DEFAULT_TEMPLATE_FLAG DEFAULT_TEMPLATE_FLAG, HRA_FEEDBACK_REQ_TMPLS_B.ALLOW_ADTNL_QSTNS_FLAG ALLOW_ADTNL_QSTNS_FLAG, HRA_FEEDBACK_REQ_TMPLS_B.GOAL_ATTRIBUTE_LIST, HRA_FEEDBACK_REQ_TMPLS_B.CUSTOM_FEEDBACK_TITLE, HRA_FEEDBACK_REQ_TMPLS_B.SHARE_WITH_MANAGERS FROM HRA_FEEDBACK_REQ_TMPLS_B, HRA_FEEDBACK_REQ_TMPLS_TL WHERE HRA_FEEDBACK_REQ_TMPLS_B.FEEDBACK_REQ_TMPL_ID = HRA_FEEDBACK_REQ_TMPLS_TL.FEEDBACK_REQ_TMPL_ID AND HRA_FEEDBACK_REQ_TMPLS_B.BUSINESS_GROUP_ID = HRA_FEEDBACK_REQ_TMPLS_TL.BUSINESS_GROUP_ID AND HRA_FEEDBACK_REQ_TMPLS_TL.LANGUAGE = USERENV('LANG')
```
