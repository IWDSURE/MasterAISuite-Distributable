## Query

```sql
SELECT AT.ANALYTIC_TYPE_ID ANALYTIC_TYPE_ID, AT.BUSINESS_GROUP_ID BUSINESS_GROUP_ID, AT.DASHBOARD_TMPL_ID DASHBOARD_TMPL_ID, AT.ANALYTIC_TYPE_CODE ANALYTIC_TYPE_CODE, AT.HORIZONTAL_AXIS_CODE HORIZONTAL_AXIS_CODE, AT.VERTICAL_AXIS_CODE VERTICAL_AXIS_CODE, AT.COLOR_SCHEME_CODE COLOR_SCHEME_CODE, AT.MODULE_ID MODULE_ID, AT.ANALYTIC_VIEW_MODE ANALYTIC_VIEW_MODE, AT.METRIC_CODE METRIC_CODE, AT.DEFAULT_VIEW_FLAG DEFAULT_VIEW_FLAG, AT.SUBMIT_BOX_ASGNMNT_FLAG SUBMIT_BOX_ASGNMNT_FLAG, AT.VIEW_ROW_COUNT VIEW_ROW_COUNT, AT.VIEW_COLUMN_COUNT VIEW_COLUMN_COUNT, AT.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, AT.CREATED_BY CREATED_BY, AT.CREATION_DATE CREATION_DATE, AT.LAST_UPDATED_BY LAST_UPDATED_BY, AT.LAST_UPDATE_DATE LAST_UPDATE_DATE, AT.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, ATTL.ANALYTIC_VIEW_NAME ANALYTIC_VIEW_NAME FROM HRR_TMPL_ANALYTIC_TYPES_B AT, HRR_TMPL_ANALYTIC_TYPES_TL ATTL WHERE AT.ANALYTIC_TYPE_ID = ATTL.ANALYTIC_TYPE_ID AND AT.BUSINESS_GROUP_ID = ATTL.BUSINESS_GROUP_ID AND ATTL.LANGUAGE = USERENV('LANG')
```
