```sql
SELECT b.SUPPT_OBJECT_ID SUPPT_OBJECT_ID, b.OBJECT_CODE OBJECT_CODE, b.FAMILY FAMILY, b.PRODUCT PRODUCT, b.OBJECT_SOURCE OBJECT_SOURCE, b.CATEGORY CATEGORY, b.REST_RESOURCE_PATH REST_RESOURCE_PATH, b.REST_RESOURCE_IDENTIFIER REST_RESOURCE_IDENTIFIER, b.REST_SUPPORTED_OPERATIONS REST_SUPPORTED_OPERATIONS, b.PARENT_SUPPT_OBJECT_ID PARENT_SUPPT_OBJECT_ID, b.MANDATORY_FLAG MANDATORY_FLAG, b.SEEDED_FLAG SEEDED_FLAG, b.API_SPEC API_SPEC, b.OBJECT_PROPERTIES OBJECT_PROPERTIES, b.SAMPLE_QUERIES SAMPLE_QUERIES, b.INTERNAL_NAME INTERNAL_NAME, b.INTERNAL_DESCRIPTION INTERNAL_DESCRIPTION, b.CREATED_BY CREATED_BY, b.CREATION_DATE CREATION_DATE, b.LAST_UPDATED_BY LAST_UPDATED_BY, b.LAST_UPDATE_DATE LAST_UPDATE_DATE, b.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, b.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, b.MODULE_ID MODULE_ID, tl.OBJECT_DISPLAY_NAME OBJECT_DISPLAY_NAME, tl.DESCRIPTION DESCRIPTION FROM FAI_SUPPT_OBJECTS_B b, FAI_SUPPT_OBJECTS_TL tl WHERE b.SUPPT_OBJECT_ID = tl.SUPPT_OBJECT_ID AND tl.LANGUAGE = USERENV('LANG')
```
