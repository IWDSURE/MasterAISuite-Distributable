## Query

```sql
Select B.ANOMALY_DEF_ID ,B.OBJECT_VERSION_NUMBER ,B.MODULE_ID ,B.DETECTION_KEY ,TL.DETECTION_MEANING ,TL.DEFAULT_REMEDY ,TL.DEFAULT_SYMPTOM ,TL.DESCRIPTION ,TL.THRESHOLD_VALUE_DEFINITION ,TL.THRESHOLD_DEFINITION ,B.SEND_TO ,B.PRIORITY ,B.THRESHOLD ,B.THRESHOLD_ON_VALUE ,B.DETECTION_GROUP ,B.DETECTION_LEVEL ,B.OVERWRITE_PRODUCT ,B.GRACE_PERIOD ,B.DETECTION_TYPE ,B.DETECTION_OBJECT ,B.CREATED_BY ,B.CREATION_DATE ,B.LAST_UPDATED_BY ,B.LAST_UPDATE_DATE ,B.LAST_UPDATE_LOGIN ,B.ENTERPRISE_ID ,B.SGUID FROM FAI_SAM_ANOMALY_DEF_B B, FAI_SAM_ANOMALY_DEF_TL TL WHERE B.ANOMALY_DEF_ID = TL.ANOMALY_DEF_ID AND TL.LANGUAGE = USERENV('LANG')
```
