## Query

```sql
SELECT b.DEEP_LINK_ID DEEP_LINK_ID, b.DEEP_LINK_CODE DEEP_LINK_CODE, b.USAGE_TYPE USAGE_TYPE, b.DESTINATION_URL DESTINATION_URL, b.DEEP_LINK_PROPERTIES DEEP_LINK_PROPERTIES, b.FAMILY FAMILY, b.PRODUCT PRODUCT, b.INTERNAL_NAME INTERNAL_NAME, b.INTERNAL_DESCRIPTION INTERNAL_DESCRIPTION, b.CREATED_BY CREATED_BY, b.CREATION_DATE CREATION_DATE, b.LAST_UPDATED_BY LAST_UPDATED_BY, b.LAST_UPDATE_DATE LAST_UPDATE_DATE, b.LAST_UPDATE_LOGIN LAST_UPDATE_LOGIN, b.OBJECT_VERSION_NUMBER OBJECT_VERSION_NUMBER, b.MODULE_ID MODULE_ID, tl.NAME NAME, tl.DESCRIPTION DESCRIPTION, tl.MESSAGE MESSAGE FROM FAI_DEEP_LINKS_B b, FAI_DEEP_LINKS_TL tl WHERE b.DEEP_LINK_ID = tl.DEEP_LINK_ID AND tl.LANGUAGE = USERENV('LANG')
```
